    <script src="https://cdnjs.cloudflare.com/ajax/libs/gsap/3.12.5/gsap.min.js"></script>
    <script src="https://cdnjs.cloudflare.com/ajax/libs/gsap/3.12.5/ScrollTrigger.min.js"></script>
    
    <script>
        gsap.registerPlugin(ScrollTrigger);
        feather.replace();

        // Reveal body after GSAP is ready
        gsap.set(".scene", { autoAlpha: 1 });

        // Modular Scene Engine
        const scenes = gsap.utils.toArray(".scene");
        
        scenes.forEach((scene, i) => {
            const content = scene.querySelector(".scene-content");
            const bgText = scene.querySelector(".bg-element");
            const frame = scene.querySelector(".visual-frame");
            
            const tl = gsap.timeline({
                scrollTrigger: {
                    trigger: scene,
                    start: "top top",
                    end: "+=150%", // Panjang scroll per scene
                    pin: true,
                    scrub: 1,
                    anticipatePin: 1
                }
            });

            // Animasi Entrance
            tl.from(content, { y: 100, opacity: 0, duration: 1 })
              .from(bgText, { x: i % 2 === 0 ? -200 : 200, opacity: 0, duration: 1.5 }, "-=1")
              .from(frame, { scale: 0.8, opacity: 0, rotation: i % 2 === 0 ? 5 : -5, duration: 1 }, "-=1");

            // Animasi Exit (Subtle fade out as we move to next)
            if (i < scenes.length - 1) {
                tl.to(content, { y: -50, opacity: 0, duration: 0.5, delay: 0.5 });
            }
        });

        // Parallax Subtle Follow (Mouse interaction optional)
        document.addEventListener("mousemove", (e) => {
            const xPos = (e.clientX / window.innerWidth - 0.5) * 20;
            const yPos = (e.clientY / window.innerHeight - 0.5) * 20;
            
            gsap.to(".visual-frame", {
                x: xPos,
                y: yPos,
                duration: 2,
                ease: "power2.out"
            });
        });
    </script>
