    <script>
    // Cinematic Hero Animation Logic
    document.addEventListener("DOMContentLoaded", () => {
        const tl = gsap.timeline({ 
            defaults: { ease: "expo.out", duration: 2 } 
        });

        // H1: Daud slide kiri, Ibrahim slide kanan
        tl.from(".hero .content h1", {
            x: -80,
            opacity: 0,
        })
        .from(".hero .content h1 span", {
            x: 160,
            opacity: 0,
        }, "<") // Start barengan biar smooth

        // Tagline: Smooth upward + blur reveal
        .fromTo(".hero .content p", 
            { 
                y: 40, 
                opacity: 0, 
                filter: "blur(12px)" 
            }, 
            { 
                y: 0, 
                opacity: 1, 
                filter: "blur(0px)",
                duration: 2.5 
            }, 
            "-=1.5" // Mulai pas H1 hampir kelar
        );
    });
</script>
