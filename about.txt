    <script>
        gsap.registerPlugin(ScrollTrigger);
        feather.replace();

        gsap.set(".scene", { autoAlpha: 1 });

        // INTRO SCROLL FIX: Logic Implementation
        (function() {
            const lightPath = document.querySelector("#fiber-path-light");
            const pathLength = lightPath.getTotalLength();

            gsap.set(lightPath, { 
                strokeDasharray: pathLength, 
                strokeDashoffset: pathLength 
            });

            const introTl = gsap.timeline({
                scrollTrigger: {
                    trigger: "#intro-sequence",
                    start: "top top",
                    end: "bottom bottom", 
                    scrub: 1, 
                }
            });

            introTl.to(lightPath, { strokeDashoffset: 0, ease: "none" })
                   .to("#intro-text", { opacity: 0, y: -40, duration: 0.5 }, "-=0.2")
                   .to(".cable-container", { opacity: 0, duration: 0.3 });
        })();

        /* MODULAR SCENE ENGINE (IDENTICAL TO PREVIOUS) */
        const allScenes = gsap.utils.toArray(".scene");
        
        allScenes.forEach((scene, i) => {
            const content = scene.querySelector(".scene-content");
            const label = scene.querySelector(".bg-label");
            const frame = scene.querySelector(".visual-frame");
            
            const sceneTl = gsap.timeline({
                scrollTrigger: {
                    trigger: scene,
                    start: "top top",
                    end: "+=120%", 
                    pin: false,
                    scrub: 1,
                    anticipatePin: 1
                }
            });

            sceneTl.from(content, { y: 60, opacity: 0, duration: 1 })
                   .from(label, { x: i % 2 === 0 ? -100 : 100, opacity: 0, duration: 1.5 }, "-=1")
                   .from(frame, { 
                        scale: 0.9, 
                        opacity: 0, 
                        rotation: i % 2 === 0 ? 3 : -3, 
                        duration: 1 
                   }, "-=1.2");

            if (i < allScenes.length - 1) {
                sceneTl.to(content, { opacity: 0, y: -40, duration: 0.5, delay: 0.3 });
            }
        });

        document.addEventListener("mousemove", (e) => {
            const xVal = (e.clientX / window.innerWidth - 0.5) * 15;
            const yVal = (e.clientY / window.innerHeight - 0.5) * 15;
            gsap.to(".visual-frame", { x: xVal, y: yVal, duration: 2.5, ease: "power2.out" });
        });
    </script>
